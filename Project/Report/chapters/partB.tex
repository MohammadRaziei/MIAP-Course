\begin{equation}\operatorname{ASSD}=\frac{\sum_{x \in \partial G} \mathbf{d}(x, \partial M)+\sum_{y \in \partial M} \mathbf{d}(y, \partial G)}{|\partial G|+|\partial M|}\end{equation}
